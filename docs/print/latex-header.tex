% --- Figure placement: keep figures inline (no floating to end) ---
\usepackage{float}
\floatplacement{figure}{H}

% --- Title page: centered, clear hierarchy for investor-facing doc ---
\makeatletter
\renewcommand{\@maketitle}{%
  \begin{center}
    \vspace*{1.5em}
    {\sffamily\LARGE\bfseries \@title \par}
    \vspace{1.25em}
    {\normalsize\normalfont \@author \par}
    \vspace{0.5em}
    {\small \@date \par}
    \vspace{2.5em}
  \end{center}
}
\makeatother

% --- Page geometry: generous margins for a professional read ---
\usepackage[a4paper, margin=1.25in, top=1.25in, bottom=1.5in]{geometry}

% --- Line spacing: improves readability ---
\usepackage{setspace}
\setstretch{1.25}

% --- Section heading style: clear hierarchy, professional look ---
\usepackage{titlesec}
\titleformat{\section}
  {\normalfont\sffamily\large\bfseries}{\thesection}{1em}{}
\titleformat{\subsection}
  {\normalfont\sffamily\normalsize\bfseries}{\thesubsection}{1em}{}
\titlespacing*{\section}{0pt}{1.5em plus 0.5em minus 0.2em}{0.8em plus 0.2em}
\titlespacing*{\subsection}{0pt}{1.2em plus 0.3em minus 0.2em}{0.5em plus 0.2em}

% --- Headers and footers: document title + page number ---
\usepackage{fancyhdr}
\pagestyle{fancy}
\fancyhf{}
\fancyfoot[C]{\sffamily\small Block Chain --- Design overview \quad \thepage}
\renewcommand{\headrulewidth}{0pt}
\fancypagestyle{plain}{\fancyhf{}\fancyfoot[C]{\sffamily\small \thepage}\renewcommand{\headrulewidth}{0pt}}

% --- Figure captions: subtle, consistent spacing ---
\usepackage[font=small, labelfont=bf, skip=8pt]{caption}

% --- Tables: slightly more readable spacing ---
